int bubblesort(student st[],int n)
{
    int i,j;
    student temp;
    for(i=0;i<n-1;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(st[i].rollno>st[j].rollno)
            {
                temp=st[j];
                st[j]=st[i];
                st[i]=temp;
            }
        }
    }
    return 0;
}int binarysearch(student st[],int n,char uname[])
{
    int i,j,mid,flag=0;
    i=0;
    j=n-1;
    while(i<=j)
    {
        mid=(i+j)/2;
        if(strcmp(uname,st[mid].name)==0)
        {
            flag=1;
            break;
        }
        else
        {
            if(strcmp(uname,st[mid].name)<0)
            {
                j=mid-1;
            }
            else
            {
                i=mid+1;
            }
        }
    }
